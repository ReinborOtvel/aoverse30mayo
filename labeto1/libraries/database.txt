pragma solidity 0.8.7;
contract database {
    address[] public accounts;
    address[] public leaders;
    string[] public statistics;

    function allAccounts() public view returns(address[] memory) {
        return accounts;
    }

    function allLeaders() public view returns(address[] memory) {
        return leaders;
    }

    function allStatistics() public view returns(string[] memory) {
        return statistics;
    }

    constructor(address _creator, string memory _statistics){
        accounts.push(_creator);
        leaders.push(_creator);
        statistics.push(_statistics);
    }

    function createAccount(address _leader, string memory _statistics) public {
        accounts.push(msg.sender);
        leaders.push(_leader);
        statistics.push(_statistics);
    }
}
