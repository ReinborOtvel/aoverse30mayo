pragma solidity 0.8.7;
contract database {
    mapping(address=>bool) public createdCharacter;
    struct Account {
        address owner;
        address leader;
        address[] group;
        string statistics;
    }
    Account[] public accounts;
    mapping(address=>uint256) public indexAccounts;
    constructor(address _account, string memory _statistics){
        createCharacter(_account,_account,_statistics);
    }
    function createCharacter(address _account,address _leader, string memory _statistics) public {
        require(createdCharacter[_account]==false);
        createdCharacter[_account]=true;
        indexAccounts[_account]=accounts.length;
        address[] memory _group;
        accounts.push(
            Account(_account,_leader,_group,_statistics)
        );
    }
    function accountsLength() public view returns(uint256) {
        return accounts.length;
    }
}
